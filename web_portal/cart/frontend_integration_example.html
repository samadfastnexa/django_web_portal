// React Example
import { useState } from 'react';

function PaymentComponent() {
  const [loading, setLoading] = useState(false);
  
  const handlePayment = async (orderId, amount) => {
    setLoading(true);
    
    try {
      const response = await fetch('http://localhost:8000/api/cart/payments/initiate/', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${token}`
        },
        body: JSON.stringify({
          order_id: orderId,
          payment_method: 'jazzcash',
          amount: amount,
          customer_phone: '03001234567'
        })
      });
      
      const result = await response.json();
      
      if (result.jazzcash_form_data) {
        // Create form and submit
        const form = document.createElement('form');
        form.method = 'POST';
        form.action = result.jazzcash_payment_url;
        
        Object.entries(result.jazzcash_form_data).forEach(([key, value]) => {
          const input = document.createElement('input');
          input.type = 'hidden';
          input.name = key;
          input.value = value;
          form.appendChild(input);
        });
        
        document.body.appendChild(form);
        form.submit();
      }
    } catch (error) {
      console.error('Payment error:', error);
    } finally {
      setLoading(false);
    }
  };
  
  return (
    <button onClick={() => handlePayment(1, 1500)}>
      {loading ? 'Processing...' : 'Pay with JazzCash'}
    </button>
  );
}
